    $unifiedOrder->setParameter("trade_type","JSAPI");//交易类型

    //$unifiedOrder->setParameter("sub_mch_id","XXXX");//子商户号

  public function log_result($file,$word)
  {
    $fp = fopen($file,"a");
    flock($fp, LOCK_EX) ;
    fwrite($fp,"执行日期：".strftime("%Y-%m-%d-%H：%M：%S",time())."\n".$word."\n\n");
    flock($fp, LOCK_UN);
    fclose($fp);
