function hide(_sId){

function onlyShow(_sId, _iNum, _sPic, _sTxt1, _sTxt2){
 var i = 0;
 var oCurr = exist(_sId + i);
 while(oCurr){

  $(_sPic + i).src = _sTxt2;
  i++;

}
