				$str_t = '';
				$len = strlen($str);
				$a = 0;
				while ($a < $len){
						if (ord($str{$a})>=224 && ord($str{$a})<=239){

										$a += 3;
										continue;
								}
						}
						$str_t .= $str{$a};
						$a += 1;
				}
