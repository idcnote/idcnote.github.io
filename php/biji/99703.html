{
  $arr = array(

"0x55","0x56","0x57","0x58","0x59","0x5A","0x5B","0x5C","0x5D","0x5E",

"0x7D","0x7E","0x7F");
  $strl = strlen($str);
  for ($i = 0;$i < $strl; $i++)
  {
    $char = '0x' . bin2hex(substr($str,$i,1));
    $pos = in_array($char,$arr);
    if ($pos == 1)
    {
      $j++;
    }
  }
  if ($j < $strl)
  {
    return false;
  }
  else
  {
    return true;
  }
}
